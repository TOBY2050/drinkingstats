  <script>

    document.addEventListener('DOMContentLoaded', () => {

      // Retrieve drinks from localStorage or initialise an empty array

      let drinks = JSON.parse(localStorage.getItem('drinks')) || [];

 

      // Update the drink counter visual

      function updateCounterVisual() {

        const counterVisual = document.getElementById('counterVisual');

        const count = drinks.length;

        counterVisual.textContent = count;

        // Scale visual up slightly for each drink (max scale of 1.5)

        const scale = 1 + Math.min(count * 0.05, 0.5);

        counterVisual.style.transform = 'scale(' + scale + ')';

      }

 

      // Function to update the drink history table

      function updateTable() {

        const tableBody = document.getElementById('drinkTableBody');

        tableBody.innerHTML = ''; // clear previous entries

        drinks.forEach((drink, index) => {

          const tr = document.createElement('tr');

          const tdIndex = document.createElement('td');

          tdIndex.textContent = index + 1;

          const tdTime = document.createElement('td');

          tdTime.textContent = drink;

          tr.appendChild(tdIndex);

          tr.appendChild(tdTime);

          tableBody.appendChild(tr);

        });

      }

 

      // Function to record a drink

      function recordDrink() {

        const now = new Date();

        const timeString = now.toLocaleString();

        drinks.push(timeString);

        localStorage.setItem('drinks', JSON.stringify(drinks));

        updateTable();

        updateCounterVisual();

 

        // Show a brief confirmation message

        const message = document.getElementById('message');

        message.textContent = 'Drink recorded at ' + timeString + '. Drink responsibly!';

        setTimeout(() => { message.textContent = ''; }, 2000);

      }

 

      // Function to export the drink data as a CSV file

      function exportData() {

        if (drinks.length === 0) {

          alert('No drinks recorded yet!');

          return;

        }

        let csvContent = "data:text/csv;charset=utf-8,Drink Number,Date & Time\n";

        drinks.forEach((drink, index) => {

          csvContent += `${index + 1},"${drink}"\n`;

        });

        const encodedUri = encodeURI(csvContent);

        const link = document.createElement('a');

        link.setAttribute('href', encodedUri);

        link.setAttribute('download', 'drink_history.csv');

        document.body.appendChild(link);

        link.click();

        document.body.removeChild(link);

      }

 

      // Function to clear history

      function clearHistory() {

        if (confirm('Are you sure you want to clear your drink history?')) {

          drinks = [];

          localStorage.setItem('drinks', JSON.stringify(drinks));

          updateTable();

          updateCounterVisual();

          const message = document.getElementById('message');

          message.textContent = 'History cleared.';

          setTimeout(() => { message.textContent = ''; }, 2000);

        }

      }

 

      // Attach event listeners to buttons

      document.getElementById('drinkBtn').addEventListener('click', recordDrink);

      document.getElementById('exportBtn').addEventListener('click', exportData);

      document.getElementById('clearBtn').addEventListener('click', clearHistory);

 

      // Initialise table and counter with any existing records

      updateTable();

      updateCounterVisual();

    });
